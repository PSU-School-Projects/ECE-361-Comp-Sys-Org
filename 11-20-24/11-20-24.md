More advanced Make file
# Make
```Make
# variable, adds the -g tag when used with gcc commands
gcc_sw = -g

# Targets
main: main.o scale.o
    # linker command
    gcc -o main.o scale.o

main.o: main.c
    gcc $(gcc_sw) -c main.o
scale.o: scale.c scale.h
    gcc $(gcc_sw) -c scale.o

clean: 
    rm main mina.o scale.o

install: main
    cp main /tmp/Mainprogram
```
